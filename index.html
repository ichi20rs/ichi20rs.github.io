import java.util.Scanner;

　　public class Main
　　{
　　 public static void main(String[] args)
{
Scanner sc = new Scanner(System.in);

int n = sc.nextInt();
int k = sc.nextInt();
int[] a = new int[n + 2];
int[] b = new int[n + 2];
int[][] dp = new int[n + 2][2];

for(int i = 1; i <= n; i++)
{
a[i] = sc.nextInt();
}
for(int i = 1; i <= n; i++)
{
b[i] = sc.nextInt();
}

dp[1][0] = 1; dp[1][1] = 1;

for(int i = 2; i <= n; i++)
{
if(dp[i-1][0] == 1 && Math.abs(a[i] - a[i-1]) <= k)
{
dp[i][0] = 1;
}
if(dp[i-1][1] == 1 && Math.abs(a[i] - b[i-1]) <= k)
{
dp[i][0] = 1;
}
if(dp[i-1][0] == 1 && Math.abs(b[i] - a[i-1]) <= k)
{
dp[i][1] = 1;
}
if(dp[i-1][1] == 1 && Math.abs(b[i] - b[i-1]) <= k) {
dp[i][1] = 1;
}

}

if(dp[n][1] == 1 || dp[n][0] == 1)
{
System.out.println("Yes");
}
else
{
System.out.println("No");
}

}
}
